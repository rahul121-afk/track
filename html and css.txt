    <script>
        // Simple script to show the tracking results when form is submitted
        document.querySelector('.search-form').addEventListener('submit', function(e) {
            e.preventDefault();
            const trackingId = document.querySelector('.search-input').value.trim();
            if (trackingId) {
                // Update tracking ID in the results
                document.querySelector('.tracking-id').textContent = trackingId;
                
                // Show results with animation
                const results = document.querySelector('.tracking-results');
                results.style.display = 'block';
                results.classList.add('fade-in');
                
                // Scroll to results
                results.scrollIntoView({ behavior: 'smooth' });
            }
        });
    </script>
